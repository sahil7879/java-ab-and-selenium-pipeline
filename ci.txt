name: Java CI/CD Pipeline

on:
  push:
    branches:
      - main
  pull_request:
    branches:
      - main

jobs:
  build:
    runs-on: ubuntu-latest

    services:
      selenium-hub:
        image: selenium/hub:4.1.2
        ports:
          - 4444:4444
      chrome:
        image: selenium/node-chrome:4.1.2
        options: --network-alias=chrome
        depends_on:
          - selenium-hub
      firefox:
        image: selenium/node-firefox:4.1.2
        options: --network-alias=firefox
        depends_on:
          - selenium-hub

    steps:
      # Step 1: Checkout the code
      - name: Checkout Code
        uses: actions/checkout@v3

      # Step 2: Set up Java environment (JDK 11 in this case)
      - name: Set up JDK 11
        uses: actions/setup-java@v3
        with:
          distribution: 'temurin' # Use the OpenJDK distribution
          java-version: '11'

      # Step 3: Cache Maven dependencies for faster builds
      - name: Cache Maven dependencies
        uses: actions/cache@v3
        with:
          path: ~/.m2
          key: ${{ runner.os }}-maven-${{ hashFiles('**/pom.xml') }}
          restore-keys: |
            ${{ runner.os }}-maven-

      # Step 4: Build and package the Java application using Maven
      - name: Build with Maven
        run: mvn clean install -B

      # Step 5: Run Unit Tests
      - name: Run Unit Tests
        run: mvn test

      # Step 6: Set up Apache Benchmark (AB)
      - name: Install Apache Benchmark
        run: sudo apt-get install -y apache2-utils

      # Step 7: Run Load Test with Apache Benchmark
      - name: Apache Benchmark Test
        run: ab -n 1000 -c 10 http://localhost:8080/ # Adjust URL and params as needed

      # Step 8: Run Selenium Tests on Selenium Grid
      - name: Run Selenium Tests
        run: |
          mvn -Dtest=YourSeleniumTestClass test # Replace with your Selenium test class

      # Step 9: Archive Test Results (JUnit, Selenium, Benchmark)
      - name: Archive Test Results
        uses: actions/upload-artifact@v3
        with:
          name: test-results
          path: target/surefire-reports/
